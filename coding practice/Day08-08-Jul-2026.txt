Day 08 - Introduction to Docker and CI/CD Pipelines


🔹 Ansible - Install Compose    (Become Admin -> Install pip -> Install Docker Compose)
- name: Install Docker Compose									-----> Just title
  hosts: all										-----> all servers
  become: yes										-----> become administrator
  tasks:
    - name: Install dependencies
      yum:
        name: "python3-pip"									-----> need the app store before downloading the app.
        state: present										-----> install if not present if there no need
    - name: Install docker-compose
      pip:
	name: docker-compose									-----> install docker compose using pip (think of it as: pip -> app store -> docker-compose -> Application)

🔹 Terraform - EC2 with Security Group and EBS volume  (Create Guard -> Allow SSH -> Create EC2 -> Attach Guard -> Attach Hard disk)
resource "aws_security_group" "allow_ssh_http" {						-----> resource "aws_security_group" MEANS Create one Security Group
  name		= "allow_ssh_http"
  description	= "Allow SSH and HTTP access"
  ingress {										-----> means Incoming Traffic. (People coming inside)
    from_port	= 22										-----> from door number 22 (SSH)
    to_port	= 22
    protocol	= "tcp"										-----> Communication language
    cidr_blocks = ["0.0.0.0/0"]										-----> everyone can enter [Interview Question: is this safe? no in production we allow only trusted IP address]
  }

  ingress {
    from_port	= 80
    to_port	= 80
    protocol	= "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_instance" "vishwesh-ec2" {							-----> Create one Computer
  ami		= "ami-0abc1234abcd1234"							-----> Amazon machine Image 
  instance_type	= "t3.medium"								-----> size of computer
  security_groups = [aws_security_group.allwo_ssh_http.name]					-----> Attach the security guard to the EC2

  ebs_block_device {										-----> Image buying a laptop with storage 10GBs now attach another 30 GBs hard disk. That's EBS.
    device_name = "/dev/sdh"
    volume_size = 30									-----> volume size 30 GB
  }
}

🔹 Kubernetes - Service with NodePort  (Outside User -> NodePort -> Service -> Pod
apiVersion: v1
kind: Service										-----> Create a service
metadata:
  name: vishwesh-service
spec: 
  type: NodePort										-----> open one port on the node. (Imagine Student outside can enter through gate, there is one gate, NodePort is that gate)
  selector: 										-----> Which Pod should receive traffic
    app: vishwesh-app										-----> Find Pods with this label
  ports:
    - protocol: TCP
      port: 80 										-----> App listens on Port 80
      nodePort: 30007									-----> Visitors connect using serverIP: 30007 Kubernetes forwards traffic to Port 80.

🔹Jenkinsfile - Docker Build and Push 		( Think of Jenkins as a robot worker. Every time new code comes... Robot does everything. Developer writes code -> Jenkins Detects -> Build Image -> Login -> Push Image)
pipeline {
    agent any
    stages {
	stage('Build Docker Image'){ 								-----> Robot builds the Docker image.
	  steps {
	      script {
		  docker.build('vishwesh-app') 							-----> Create the Image
	      }
           }
        }
     }
     stages('Push Docker Image') { 								-----> Robot uploads the image
	  steps {
	      script {
		  docker.withRegistry('https://docker.in', 'docker-credentials') { 		-----> Login to Docker Registry
		     docker.image('vishwesh-app').push('latest') 				-----> push('latest') means upload image
	      }
           }
        }
     }
  }
}

🔹 GitLab CI/CD - Build & Push Docker Image (GitLab CI/CD does almost the same job as Jenkins. Instead of Jenkins, Gitlab perform the Automation. CODE -> BUILD -> LOGIN -> PUSH)
stages:
  - build 										-----> Build Docker Image
  - push 										-----> push Doker image

build:
  stage: build
  script: 
    - docker build -t vishwesh-app

push:
  stages: push
  script:
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD" 				-----> login
    - docker push vishwesh-app 									-----> upload image
