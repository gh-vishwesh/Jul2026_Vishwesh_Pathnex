🔹Ansible - Create User and Set Permissions
- name: Create a new user
  hosts: all
  become: yes
  tasks: 
    - name: Create user
      user:
        name: vishwesh
	state: present
	shell: /bin/bash
    - name: Set permissions for the user
      file:
	path: /home/vishwesh
	owner: vishwesh
	group: vishwesh
	mode: '0755'
