  name			= "vishwesh-lb"
  internal		= false							----> false means public can access it. internal = true means can't

  subnets		= [aws_subnet.public.id]				----> which subnet should contain the Load Balancer?
}

apiVersion: appsv1
kind: Deployment
metadata:
  name: vishwesh-web-app
spec:
  replicas: 2									----> start with 2 pods
  selector: 
    matchLabels:
      app: vishwesh-web-app
  template:
    metadata:
      labels:
        app: vishwesh-web-app
    spec:
      containers:
	- name: web
	  image: nginx
	  ports:
	    - containerPort: 80

---
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler							----> Automatically increase Pods
metadata:
  name: vishwesh-web-app-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: vishwesh-web-app
  minReplicas: 1								----> minimum one pod
  maxReplicas: 5								----> maximum 5 pods
  targetCPUUtilizationPercentage: 80						----> if CPU usage reaches around 80%, create more pod

🔹Jenkinsfile - Implement Parallel Stages
pipeline {
    agent any
    stages {
        stage('Build & Test') {
	  parallel {								----> run multiple stages at the same time
	      stage('Build') {							----> build application
		steps {
		    echo 'Building project...'
		}
	      }
	      stage('Test') {							----> run test
                    steps {
                        echo 'Running tests...'
                    }
                }
            }
        }
        stage('Deploy') {							----> then deploy only after both (build + test) finishes successfully.
            steps {
                echo 'Deploying project...'
            }
        }
    }
}

[Why use parallel stages?
Answer:To reduce pipeline execution time by running independent tasks simultaneously.]
