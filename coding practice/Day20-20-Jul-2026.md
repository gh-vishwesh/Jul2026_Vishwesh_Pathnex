🔹 Ansible - Configure Fireball with iptables                                     
- name: Configure Fireball with iptables
  hosts: all
  become: yes
  tasks:
    - name: Set default policies to drop all traffic
      command: iptables -P INPUT DROP                                       # → Sets the default INPUT policy to DROP all incoming traffic.
    - name: Allow SSH and HTTP traffic
      command: iptables -A INPUT -p tcp --dport 22 -j ACCEPT                # → Allows incoming SSH traffic on port 22.
      command: iptables -A INPUT -p tcp --dport 80 -j ACCEPT                # → Allows incoming SSH traffic on port 80.
    - name: Save iptables configuration                                     # → Names the task to save firewall rules.
      command: service iptables save                                        # → Saves the current iptables configuration.

🔹 Terraform - Configure Security Groups for EC2
resource "aws_security_group" "vishwesh_sg" {                               # → Creates an AWS Security Group.
  name_prefix = "vishwesh-sg"
  description = "Allow SSH and HTTP"

  ingress {                                                                 # → Starts an inbound traffic rule.
    from_port    = 22                                                       # → Opens inbound traffic from port 22.
    to_port = 22                                                            # → Ends the allowed port range at 22.
    protocol = "tcp"                                                        # → Allows TCP traffic.
    cidr_blocks = ["0.0.0.0/0"]                                             # → Allows traffic from any IP address.
  }

  ingress {
    form_port   = 80
    to_port = 80
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

🔹Kubernetes - RBAC Configuration for Access Control

metadata:
  name: vishwesh-clustion-admin

  - kind: User                                                              # → Grants permissions to a Kubernetes user.
    name: vishwesh-user                                                     # → Specifies the user name.
    apiGroup: rbac.authorization.k8s.io                                     # → Uses the RBAC API group.
roleRef:                                                                    # → Specifies which role is assigned.
  kind: ClusterRole                                                         # → References a ClusterRole.
  name: cluster-admin                                                       # → Grants the cluster-admin role.
  apiGroup: rbac.authorization.k8s.io                                       # → References the RBAC API group.

🔹Jenkinsfile - Secure Credentials Management
pipeline {
    agent any
    environment {
        REGISTRY_PASSWORD = credentials('docker-registry-password')
    }
    stages {
        stage('Build') {
            steps {
                echo 'Building Docker image...'
                docker.Build('vishwesh-web-app')
            }
        }
        stage('Push') {
            steps {
                script {
                    docker.withRegistry('https://docker.io', 'docker-credentials') {
                        docker.image('vishwesh-web-app').push('latest')
                    }
                }
            }
        } 
    }
}

  stage: deploy
