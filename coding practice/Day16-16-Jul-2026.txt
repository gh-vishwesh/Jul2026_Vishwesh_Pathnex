🔹Kubernetes - Multi-Cluster Deployment with Federation
apiVersion: apps/v1											→ Uses the Kubernetes Apps API.
kind: Deployment											→ Creates a Deployment resource.
metadata:												→ Stores Deployment metadata.
  name: vishwesh-deployment										→ Names the Deployment.
  labels: 												→ Adds labels for identification.
    app: vishwesh-app											→ Labels the application.
spec: 													→ Defines the Deployment configuration.
  replicas: 2												→ Runs two Pod replicas.
  selector:												→ Selects Pods using labels.
    matchLabels:											→ Matches Pods with the specified label.
      app: vishwesh-app
  template: 												→ Defines the Pod template.
    metadata:
      labels:
	app: vishwesh-app
    spec:
      containers: 											→ Starts the container definition.
        - name: nginx											→ Names the container.
   	  image: nginx											→ Pulls and runs the Nginx image.
	  ports:											→ Defines the container ports.
	    - containerPort: 80										→ Exposes port 80 inside the container.

---

apiVersion: v1												→ Uses the Kubernetes Core API.
kind: Service												→ Creates a Kubernetes Service.
metadata: 												→ Stores Service metadata.
  name: vishwesh-service										→ Names the Service.
spec:													→ Defines the Service configuration.
  selector:												→ Selects Pods with matching labels.
    app: vishwesh-app											→ Routes traffic to Pods with this label.
  ports:												→ Defines the Service ports.
    - protocol: TCP											→ Uses the TCP protocol.
      port: 80												→ Exposes the Service on port 80.
🔹Jenkinsfile - Deploy to Multi-Cluster Environments
pipeline {												→ Starts the Jenkins pipeline.
    agent any												→ Runs the pipeline on any available Jenkins agent.
    stages {												→ Starts the pipeline stages.
	stage('Build') {										→ Defines the build stage.
	    steps {
		echo 'Building Docker image...'								→ Prints the build message.
		docker.build('vishwesh-web-app')							→ Builds the Docker image.
	    }
	}
	stage('Test') {											→ Defines the testing stage.
	    steps {
		echo 'Running tests...'									→ Prints the testing message.
	    }
	}
	stage('Deploy to Kubernetes') {									→ Defines the deployment stage.
	    steps {
		script {										→ Executes conditional Groovy code.
		    if (env.CLUSTER == 'us-east') {							→ Checks whether the target cluster is US East.
			sh 'kubectl --context=us-east apply -f kubernetes/deployment.yaml'		→ Deploys to the US East Kubernetes cluster.
		    } else {										→ Executes when the target cluster is not US East.
			sh 'kubectl --context=us-west apply -f kubernetes/deployment.yaml'		→ Deploys to the US West Kubernetes cluster.
		    }
		}
	     }
	}
    }
}

🔹GitLab CI/CD - Multi-Region Deployment with Kubernetes
stages:													→ Defines the CI/CD pipeline stages.
  - build												→ First stage builds the application.
  - push												→ Second stage pushes the Docker image.
  - deploy												→ Third stage deploys the application.

build:													→ Starts the build job.
  stage: build
  script: 
    - docker build -t vishwesh-web-app .								→ Builds the Docker image.

push:													→ Starts the image push job.
  stage: push
  script: 
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"					→ Logs in to the Docker registry.
    - docker push vishwesh-web-app									→ Pushes the Docker image.

deploy:													→ Starts the deployment job.
  stage: deploy
  script: 
    - |
      if [ "$CI_COMMIT_REF_NAME" == "main" ]; then							→ Checks whether the branch is main.
	kubectl --context=us-east apply -f Kubernetes/prod-deployment.yaml				→ Deploys to the US East Kubernetes cluster.
      else												→ Executes for all other branches.
	kubectl --context=us-west apply -f Kubernetes/prod-deployment.yaml				→ Deploys to the US West Kubernetes cluster.
      fi

🔹 Docker
# Shell Script Execution
#!/bin/bash												→ Specifies the Bash shell interpreter.
echo "Hello Vishwesh"											→ Prints Hello Vishwesh when the script runs.

# Docker File
FROM ubuntu												→ Uses Ubuntu as the base image.
WORKDIR /opt/vishwesh/scripts										→ Sets the container's working directory.
COPY start.sh /opt/vishwesh/scripts/start.sh								→ Copies the shell script into the container.
CMD ["/opt/vishwesh/scripts/start.sh"]									→ Executes the shell script when the container starts.
